 80def map_frequency(diff):
 81    """Map timedelta to frequency string"""
 82    days = diff.days
 83    if days > 0 and days <= 3:
 84        return "D"
 85    elif days > 3 and days <= 14:

 87    elif days > 14 and days <= 60:
 88        return "MS"
 89    elif days > 60 and days <= 120:

123def convert_metric_prefix(value_str: str) -> float:
124    """Convert string with metric prefix to float
125    e.g., '1.3K' -> 1300, '2.6M' -> 2600000
126    """
127    # Dictionary of metric prefixes and their multipliers
128    metric_prefixes = {
129        'K': 1000,
130        'M': 1000000, 
131        'B': 1000000000,
132        'G': 1000000000,
133        'T': 1000000000000}
134    
135    # Clean and standardize input
136    value_str = value_str.strip().upper()
137    
138    try:
139        # Match number and prefix
140        match = re.match(r'^(-?\d*\.?\d+)([KMGBT])?$', value_str)
141        if match:
142            number, prefix = match.groups()
143            number = float(number)
144            # Multiply by prefix value if present
145            if prefix and prefix in metric_prefixes:
146                number *= metric_prefixes[prefix]
147            return number
148        return float(value_str)  # No prefix case
149    except:
150        print(f"Error converting value: {value_str}")
151        return value_str  # Return original if conversion fails

167    """

169
170    Parameters:

175    Returns:
176    pd.Series: The normalized pandas Series.
177    """
178    series_min = series.min()
179    series_max = series.max()
180    normalized_series = (series - series_min) / (series_max - series_min) * (new_max - new_min) + new_min

198def find_zero_crossing(series):
199    """Find the x-intercept (value where series crosses zero) using linear interpolation"""
200    if not ((series.min() < 0) and (series.max() > 0)):
201        return None
202        
203    # Find where values change sign
204    for i in range(len(series)-1):
205        if (series.iloc[i] <= 0 and series.iloc[i+1] > 0) or \
206           (series.iloc[i] >= 0 and series.iloc[i+1] < 0):
207            # Get x values (index values)
208            x1, x2 = series.index[i], series.index[i+1]
209            # Get y values
210            y1, y2 = series.iloc[i], series.iloc[i+1]
211            # Linear interpolation to find x-intercept
212            zero_x = x1 + (0 - y1)*(x2 - x1)/(y2 - y1)
213            return zero_x
214            
215    return None

217def round_to_month_start(dates: pd.DatetimeIndex):
218    """Round dates to nearest month start.
219    
220    Args:
221        dates: DatetimeIndex of dates to round
222        
223    Returns:
224        DatetimeIndex: Dates rounded to nearest month start
225    """
226    def _round_single_date(dt):
227        if dt.day > 15:
228            # Roll forward to next month start
229            if dt.month == 12:
230                return pd.Timestamp(f"{dt.year + 1}-01-01")
231            else:
232                return pd.Timestamp(f"{dt.year}-{dt.month + 1:02d}-01")
233        else:
234            # Roll back to current month start
235            return pd.Timestamp(f"{dt.year}-{dt.month:02d}-01")
236
237    rounded_dates = [_round_single_date(dt) for dt in dates]
238    return pd.DatetimeIndex(rounded_dates)

241    """Round dates to nearest frequency start.

243    Args:
244        dates: DatetimeIndex to round
245        freq: Frequency string (e.g. 'MS', 'QS', 'AS', 'W-SUN')
246    
247    Returns:
248        DatetimeIndex: Dates rounded to nearest frequency start
249    """
250    # Map common frequencies to period frequencies
251    freq_map = {
252        'MS': 'M',  # Month Start -> Month
253        'QS': 'Q',  # Quarter Start -> Quarter
254        'AS': 'A',  # Year Start -> Year
255        'W-SUN': 'W'  # Week Start (Sunday) -> Week
256    }
257    
258    # Convert frequency if needed
259    period_freq = freq[0] #Try just taking the first letter of the frequency string.
260    #freq_map.get(freq, freq)
261    
262    def _round_single_date(dt):
263        # Get current and next period starts
264        current_period = pd.Timestamp(dt).normalize().to_period(period_freq).to_timestamp()
265        next_period = (pd.Timestamp(dt).normalize().to_period(period_freq) + 1).to_timestamp()
266        
267        # Calculate distances
268        dist_to_current = abs((dt - current_period).total_seconds())
269        dist_to_next = abs((dt - next_period).total_seconds())
270        
271        # Return closest period start
272        return next_period if dist_to_next < dist_to_current else current_period

274    rounded_dates = [_round_single_date(dt) for dt in dates]
